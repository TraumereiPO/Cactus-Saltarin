var direccion = Vector2.LEFT
@warning_ignore("unused_parameter")

	velocity = direccion * speed

	speed *= factor
	if $AnimatedSprite2D.animation != "correr": # Cambia la animaci√≥n del dinosaurio
		$AnimatedSprite2D.play("correr")
